 11 
 12 let points = [];
 13 
 14 let hull = [];
 15 
 16 /**
 17 *	Enumeration of the possible state the jarvis march algorithm can be in
 18 *
 19 *	@enum {Integer}
 20 */
 21 const jarvisSteps = Object.freeze({
 22 	LEFT: 0,
 23 	CALCULATE: 1,
 24 	DONE: 2
 25 });
 26 
 27 let jarvisStep = 0;
 28 
 29 let leftPointIndex = 0;;
 30 
 31 let currentIndex = -1;
 32 
 33 let nextIndex = 1;
 34 
 35 let index = 2;
 36 
 37 let time = 0;
 38 
 39 const leftPointFrameRate = 10;
 40 
 41 const calculateFrameRate = 15;
 42 
 43 /**
 44 *	Enumeration of the possible directions the algorithm can use when
 45 *	generating the convex hull
 46 *
 47 *	@enum {String}
 48 */
 49 const direction = Object.freeze({
 50 	CLOCKWISE: "clockwise",
 51 	ANTICLOCKWISE: "anticlockwise"
 52 });
 53 
 54 let angularDirection = direction.CLOCKWISE;
 55 
 56 
 57 /**
 58 *   Jarvis March algorithm function
 59 */
 60 function jarvisMarch() {
 61 	switch(jarvisStep) {
 62 
 63 		// find left most point
 64 		case jarvisSteps.LEFT:
 65 			if(currentIndex != -1) {
 66 				if(points[currentIndex].x < points[leftPointIndex].x) {
 67 					points.sort((a, b) => a.y - b.y);
 68 					leftPointIndex = currentIndex;
 69 					frameRate(leftPointFrameRate);
 70 				}
 71 
 72 				// Draw red around the point that is being checked
 73 				push();
 74 				stroke(255, 0, 0);
 75 				strokeWeight(0.3 * pointRadius);
 76 				const currentPoint = points[currentIndex];
 77 				line(0, currentPoint.y, currentPoint.x, currentPoint.y);
 78 				ellipse(currentPoint.x, currentPoint.y, pointRadius, pointRadius);	
 79 				pop();
 80 
 81 				// Draw green around the current left most point
 82 				push();
 83 				stroke(0, 255, 0);
 84 				strokeWeight(0.3 * pointRadius);
 85 				const leftPoint = points[leftPointIndex];
 86 				line(0, leftPoint.y, leftPoint.x, leftPoint.y);
 87 				ellipse(leftPoint.x, leftPoint.y, pointRadius, pointRadius);	
 88 				pop();
 89 			} else {
 90 				currentIndex = 0;
 91 				leftPointIndex = currentIndex;
 92 			}
 93 
 94 			currentIndex++;
 95 			if(currentIndex == points.length) {
 96 				jarvisStep++;
 97 				points.sort((a, b) => a.x - b.x);
 98 				leftPointIndex = 0;
 99 				currentIndex = leftPointIndex;
100 				nextIndex = 1;
101 				index = 2;
102 				hull.push(points[leftPointIndex].copy());
103 				frameRate(calculateFrameRate);
104 			}
105 		break;
106 
107 		// calculate which points are in the hull
108 		case jarvisSteps.CALCULATE:
109 			drawHull(hull, 65);
110 
111 			const nextPoint = points[nextIndex];
112 			const currentPoint = points[currentIndex];
113 			const checking = points[index];
114 
115 			push();
116 			stroke(0, 255, 0);
117 			line(currentPoint.x, currentPoint.y, checking.x, checking.y);
118 			strokeWeight(3);
119 			stroke(255, 0, 0);
120 			line(currentPoint.x, currentPoint.y, nextPoint.x, nextPoint.y);
121 			pop();
122 			const a = p5.Vector.sub(nextPoint, currentPoint);
123 			const b = p5.Vector.sub(checking, currentPoint);
124 			const cross = a.cross(b);
125 
126 			const clockwiseCheck = angularDirection == direction.CLOCKWISE && cross.z < 0;
127 			const anticlockwiseCheck = angularDirection == direction.ANTICLOCKWISE && cross.z > 0;
128 
129 			if ( clockwiseCheck || anticlockwiseCheck) {
130 				nextIndex = index;
131 			} 
132 			index++;
133 
134 			if (index == points.length) {
135 			    if (nextIndex == 0) {
136 			    	jarvisStep = jarvisSteps.DONE;
137 			    	time = 0;
138 			    } else {
139 			        hull.push(points[nextIndex]);
140 			        currentIndex = nextIndex;
141 			        index = 0;
142 			        nextIndex = leftPointIndex;
143 			    }
144 			}
145 		break;
146 
147 		// Draw Final hull for 5 seconds then restart
148 		case jarvisSteps.DONE:
149 			drawHull(hull, 65);
150 			time++;
151 			if (time > frameRate() * 4) {
152 				reset();
153 			}
154 		break;
155 
156 	}
157 }
