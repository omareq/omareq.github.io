 12 /**
 13 *	Enumeration of the possible state the graham scan algorithm can be in
 14 *
 15 *	@enum {Integer}
 16 */
 17 const grahamSteps = Object.freeze({
 18 	SORT: 0,
 19 	CALCULATE: 1,
 20 	DONE: 2
 21 });
 22 
 23 let grahamStep = grahamSteps.SORT;
 24 
 25 
 26 /**
 27 *	The graham scan algorithm
 28 */
 29 function grahamScan() {
 30 	switch(grahamStep) {
 31 		case grahamSteps.SORT:
 32 		points.sort((a, b) => a.y - b.y);
 33 		const lowestPoint = points[0];
 34 		const lpx = lowestPoint.x;
 35 		const lpy = lowestPoint.y;
 36 		points.sort((a, b) => atan2(a.y - lpy, a.x - lpx) - atan2(b.y - lpy, b.x - lpx));
 37 
 38 		push();
 39 		fill(255, 0, 0);
 40 		ellipse(lowestPoint.x, lowestPoint.y, pointRadius, pointRadius);
 41 
 42 		stroke(255);
 43 		strokeWeight(4);
 44 		line(lowestPoint.x, lowestPoint.y, points[1].x, points[1].y);
 45 		line(lowestPoint.x, lowestPoint.y, points[points.length - 1].x, points[points.length - 1].y);
 46 
 47 		index++;
 48 		if(index == points.length) {
 49 			index = 0;
 50 			grahamStep++;
 51 			hull.push(points[0]);
 52 			hull.push(points[1]);
 53 			index = 2;
 54 			return;
 55 		}
 56 
 57 		stroke(0, 255, 100);
 58 		strokeWeight(2);
 59 		line(lowestPoint.x, lowestPoint.y, points[index].x, points[index].y);
 60 
 61 		pop();
 62 		break;
 63 
 64 		case grahamSteps.CALCULATE:
 65 			drawEdges(hull, 55);
 66 
 67 	    	while (hull.length >= 2 && crossZ(hull[hull.length - 2], hull[hull.length - 1], points[index]) <= 0) {
 68 	        	hull.pop();
 69 	      	}
 70 	    	hull.push(points[index]);
 71 
 72 		    index++;
 73 	      	if(hull[hull.length - 1] == points[points.length - 1]) {
 74 		    	grahamStep++;
 75 		    	index = 0;
 76 		    	console.log("Done");
 77 			}
 78 		break;
 79 
 80 		case grahamSteps.DONE:
 81 			drawHull(hull, 55);
 82 			time++;
 83 			if (time > frameRate() * 4) {
 84 				grahamStep = grahamSteps.SORT;
 85 				reset();
 86 			}
 87 		break
 88 	}
 89 	return;
 90 }
