 12 let internalHulls = [[], [], []];
 13 
 14 let internalPoints = [[], [], []];
 15 
 16 let finalHull = [];
 17 let finalPoints = [];

 19 /**
 20 *	Enumeration of the possible state the algorithm can be in
 21 *
 22 *	@enum {Integer}
 23 */
 24 const divideSteps = Object.freeze({
 25 	SPLIT: 0,
 26 	CALCULATE1: 1,
 27 	CALCULATE2: 2,
 28 	CALCULATE3: 3,
 29 	FUSE: 4,
 30 	DONE: 5
 31 });
 32 
 33 let divideStep = divideSteps.SPLIT;
 34 
 35 /**
 36 *	Enumeration of the different methods for splitting the points
 37 *
 38 *	@enum {String}
 39 */
 40 const splitMethods = Object.freeze({
 41 	VERTICAL: " Vertical Split ",
 42 	HORIZONTAL: " Horizontal Split ",
 43 	RADIAL: " Radial Split ",
 44 	ANGULAR: " Angular Split "
 45 });

 47 let splitMethod = splitMethods.HORIZONTAL;

 49 /**
 50 *	Returns an array with arrays of the given size.
 51 *
 52 *	@param array {Array} Array to split
 53 *
 54 *	@param chunkSize {Integer} Size of every group
 55 */
 56 function chunkArray(inputArray, chunkSize) {
 57 	// ES6 Clone Array
 58     let array = [...inputArray];
 59     let results = [];
 60 
 61     while (array.length) {
 62         results.push(array.splice(0, chunkSize));
 63     }
 64 
 65     let extra = [];
 66     if(results[results.length - 1].length != results[results.length - 2].length) {
 67     	extra = results.pop();
 68     }
 69 
 70     while(extra.length) {
 71     	results[results.length - 1].push(extra.pop());
 72     }
 73 
 74     return results;
 75 }

 77 /**
 78 *	Draws a convex hull with a given colour
 79 *
 80 *	@param hullArray {Array<p5.Vector>} Array of points that amke up the
 81 *		verticies of the hull.
 82 *	
 83 *	@param colour {Integer} Value of the hue of the given hull from 0 to 100
 84 */
 85 function drawHull(hullArray, colour) {
 86 	push();
 87 	colorMode(HSB, 100);
 88 	fill(colour, 100, 100, 50);
 89 	beginShape();
 90 	for (let i = hullArray.length - 1; i >= 0; i--) {
 91 		push();
 92 		stroke(colour, 100,100);
 93 		strokeWeight(0.3 * pointRadius);
 94 		ellipse(hullArray[i].x, hullArray[i].y, pointRadius, pointRadius);
 95 		vertex(hullArray[i].x, hullArray[i].y);
 96 		pop();
 97 	}
 98 	endShape(CLOSE);
 99 	colorMode(RGB);
100 	pop();
101 }

103 /**
104 *	Calculates the internal hull for one of the subsections after division
105 *
106 *	@param hullIndex {Integer} Select which hull to calculate from 0 to 2
107 */
108 function calculateHull(hullIndex) {
109 	frameRate(calculateFrameRate);			
110 
111 	const currentPoint = internalPoints[hullIndex][currentIndex];
112 	const nextPoint = internalPoints[hullIndex][nextIndex];
113 	const checking = internalPoints[hullIndex][index];
114 
115 	push();
116 	stroke(0, 255, 0);
117 	line(currentPoint.x, currentPoint.y, checking.x, checking.y);
118 	strokeWeight(3);
119 	stroke(255, 0, 0);
120 	line(currentPoint.x, currentPoint.y, nextPoint.x, nextPoint.y);
121 	pop();
122 	const a = p5.Vector.sub(nextPoint, currentPoint);
123 	const b = p5.Vector.sub(checking, currentPoint);
124 	const cross = a.cross(b);
125 
126 	if (cross.z < 0) {
127 		nextIndex = index;
128 	}
129 	index++;
130 
131 	if (index == internalPoints[hullIndex].length) {
132 	    if (nextIndex == 0) {
133 	    	divideStep++;
134 	    	currentIndex = 0;
135 			nextIndex = 1;
136 			index = 2;
137 			if(hullIndex + 1 != internalPoints.length) {
138 				internalHulls[hullIndex + 1].push(internalPoints[hullIndex + 1][0].copy());
139 	    	}
140 	    } else {
141 	        internalHulls[hullIndex].push(internalPoints[hullIndex][nextIndex]);
142 	        currentIndex = nextIndex;
143 	        index = 0;
144 	        nextIndex = 0;
145 	    }
146 	}		
147 }

149 /**
150 *	Combines the three internal hulls into the final convex hull for all points
151 */
152 function fuse() {
153 	if(finalPoints.length == 0) {
154 		finalPoints = [...internalHulls[0]].concat(internalHulls[1]).concat(internalHulls[2]);
155 
156 		finalPoints.sort((a, b) => a.x - b.x);
157 		finalHull.push(finalPoints[0]);
158 	} else {
159 		frameRate(calculateFrameRate + 10);			
160 
161 		const currentPoint = finalPoints[currentIndex];
162 		const nextPoint = finalPoints[nextIndex];
163 		const checking = finalPoints[index];
164 
165 		push();
166 		stroke(0, 255, 0);
167 		line(currentPoint.x, currentPoint.y, checking.x, checking.y);
168 		strokeWeight(3);
169 		stroke(255, 0, 0);
170 		line(currentPoint.x, currentPoint.y, nextPoint.x, nextPoint.y);
171 		pop();
172 		const a = p5.Vector.sub(nextPoint, currentPoint);
173 		const b = p5.Vector.sub(checking, currentPoint);
174 		const cross = a.cross(b);
175 
176 		if (cross.z < 0) {
177 			nextIndex = index;
178 		}
179 		index++;
180 
181 		if (index == finalPoints.length) {
182 		    if (nextIndex == 0) {
183 		    	divideStep++;
184 		    	currentIndex = 0;
185 				nextIndex = 1;
186 				index = 2;
187 				time = 0;
188 		    } else {
189 		        finalHull.push(finalPoints[nextIndex]);
190 		        currentIndex = nextIndex;
191 		        index = 0;
192 		        nextIndex = 0;
193 		    }
194 		}
195 	}
196 }

198 /**
199 *	The Divide & Conquer algorithm
200 */
201 function divideAndConquer() {
202 	switch(divideStep) {
203 		case divideSteps.SPLIT:
204 			switch(splitMethod) {
205 				case splitMethods.VERTICAL: {
206 					points.sort((a, b) => a.x - b.x);
207 					internalPoints = chunkArray(points, points.length / 3);
208 				}
209 				break;
210 
211 				case splitMethods.HORIZONTAL: {
212 					points.sort((a, b) => a.y - b.y);
213 					internalPoints = chunkArray(points, points.length / 3);
214 				}
215 				break;
216 
217 				case splitMethods.RADIAL: {
218 					const w2 = width / 2;
219 					const h2 = height / 2;
220 					points.sort((a, b) => atan2(a.y - h2, a.x - w2) - atan2(b.y - h2, b.x - w2));
221 					internalPoints = chunkArray(points, points.length / 3);
222 				}
223 				break;
224 
225 				case splitMethods.ANGULAR: {
226 					const w2 = width / 2;
227 					const h2 = height / 2;
228 					points.sort((a, b) => dist(a.x, a.y, w2, h2) - dist(b.x, b.y, w2, h2));
229 					internalPoints = chunkArray(points, points.length / 3);
230 				}
231 				break;
232 			}
233 			
234 			currentIndex = 0;
235 			nextIndex = 1;
236 			index = 2;
237 			internalHulls[0].push(internalPoints[0][0].copy());
238 			divideStep++;
239 		break;
240 
241 		case divideSteps.CALCULATE1:
242 			drawHull(internalHulls[0], 0);
243 			calculateHull(0);
244 		break;
245 
246 		case divideSteps.CALCULATE2:
247 			drawHull(internalHulls[0], 0);
248 			drawHull(internalHulls[1], 30);
249 			calculateHull(1);
250 		break;
251 
252 		case divideSteps.CALCULATE3:
253 			drawHull(internalHulls[0], 0);
254 			drawHull(internalHulls[1], 30);
255 			drawHull(internalHulls[2], 60);
256 			calculateHull(2);
257 		break;
258 
259 		case divideSteps.FUSE:
260 			drawHull(internalHulls[0], 0);
261 			drawHull(internalHulls[1], 30);
262 			drawHull(internalHulls[2], 60);
263 			
264 			fuse();
265 			drawHull(finalHull, 90);
266 		break;
267 
268 		case divideSteps.DONE:
269 			drawHull(finalHull, 90);
270 			time++;
271 
272 			if (time > frameRate() * 4) {
273 				divideStep = divideSteps.SPLIT;
274 				time = 0;
275 				points = [];
276 				finalHull = [];
277 				finalPoints = [];
278 				internalHulls = [[], [], []];
279 				internalPoints = [[], [], []];
280 				currentIndex = 0;
281 				nextIndex = 1;
282 				index = 2;
283 
284 				for(let i = 0; i < numPoints; ++i) {
285 					const x = random(xBuffer, width - xBuffer);
286 					const y = random(yBuffer, height - yBuffer);
287 					const newPoint = createVector(x, y);
288 					points.push(newPoint);
289 				}
290 			}
291 		break;
292 	}
293 }
