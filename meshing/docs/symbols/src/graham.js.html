 12 /**
 13 *	Finds the z component of the cross product of three vectors
 14 *
 15 *	@param a {p5.Vector}
 16 *
 17 *	@param b {p5.Vector}
 18 *
 19 *	@param o {p5.Vector}
 20 *
 21 *	@returns {Float} 
 22 */
 23 function crossZ(a, b, o) {
 24 	let va = p5.Vector.sub(a, o);
 25 	let vb = p5.Vector.sub(b, o);
 26 	let c = va.cross(vb);
 27 	return c.z;
 28 }

 30 /**
 31 *	The graham scan algorithm
 32 */
 33 function grahamScan(verticies) {
 34 	let hull = [];
 35 	let points = [...verticies];
 36 
 37 	points.sort((a, b) => a.y - b.y);
 38 	const lowestPoint = points[0];
 39 	const lpx = lowestPoint.x;
 40 	const lpy = lowestPoint.y;
 41 	points.sort((a, b) => atan2(a.y - lpy, a.x - lpx) - atan2(b.y - lpy, b.x - lpx));
 42 
 43 	hull.push(points[0]);
 44 	hull.push(points[1]);
 45 	let index = 2;
 46 
 47 	for(;;) {
 48     	while (hull.length >= 2 && crossZ(hull[hull.length - 2], hull[hull.length - 1], points[index]) <= 0) {
 49         	hull.pop();
 50       	}
 51     	hull.push(points[index]);
 52 
 53 	    index++;
 54       	if(hull[hull.length - 1] == points[points.length - 1]) {
 55 	    	return hull;
 56 		}
 57 	}
 58 }
