 12 /**
 13 *	Aeray storing a list of randomly generated points
 14 *
 15 *	@type {Array<p5.Vector>}
 16 */
 17 let myPoints = [];
 18 
 19 /**
 20 *	Number of points to be bound by the generated convex hull
 21 *
 22 *	@type {Integer}
 23 */
 24 let numPoints = 10;
 25 
 26 /**
 27 *	Size of the points drawn on the canvas
 28 *
 29 *	@type {Integer}
 30 */
 31 let pointRadius = 4;
 32 
 33 /**
 34 *	Variable that store how far away from the side edges new points can spawn
 35 *
 36 *	@type {Integer}
 37 */
 38 let xBuffer;
 39 
 40 /**
 41 *	Variable that store how far away from the top and bottom edges new points
 42 *	can spawn
 43 *
 44 *	@type {Integer}
 45 */
 46 let yBuffer;
 47 
 48 /**
 49 *	Draws a set of points on the screen
 50 *
 51 *	@param verticies {Array<p5.Vector>} Array of points
 52 */
 53 function drawPoints(verticies) {
 54 	push();
 55 	noStroke();
 56 	fill(0, 155);
 57 	let points = [...verticies];
 58 	for (var i = points.length - 1; i >= 0; i--) {
 59 		ellipse(points[i].x, points[i].y, pointRadius, pointRadius);
 60 	}
 61 	pop();
 62 }
 63 
 64 /**
 65 *	Draws a convex hull with a given colour
 66 *
 67 *	@param hullArray {Array<p5.Vector>} Array of points that amke up the
 68 *		verticies of the hull.
 69 *	
 70 *	@param colour {Integer} Value of the hue of the given hull from 0 to 100
 71 */
 72 function drawHull(hullArray, colour) {
 73 	push();
 74 	colorMode(HSB, 100);
 75 	fill(colour, 100, 100, 0);
 76 	beginShape();
 77 	for (let i = hullArray.length - 1; i >= 0; i--) {
 78 		push();
 79 		stroke(colour, 100,100);
 80 		strokeWeight(0.3 * pointRadius);
 81 		ellipse(hullArray[i].x, hullArray[i].y, pointRadius, pointRadius);
 82 		vertex(hullArray[i].x, hullArray[i].y);
 83 		pop();
 84 	}
 85 	endShape(CLOSE);
 86 	colorMode(RGB);
 87 	pop();
 88 }
 89 
 90 
 91 /**
 92 *   p5.js setup function, creates canvas.
 93 */
 94 function setup() {
 95 	let cnvSize;
 96 	if(windowWidth > windowHeight) {
 97 		cnvSize = windowHeight;
 98 	} else {
 99 		cnvSize = windowWidth;
100 	}
101 	let cnv = createCanvas(cnvSize, .7 * cnvSize);
102 	cnv.parent('sketch');
103 
104 
105 	xBuffer = 0.05 * width;
106 	yBuffer = 0.05 * height;
107 
108 	for(let i = 0; i < numPoints; ++i) {
109 		const x = random(xBuffer, width - xBuffer);
110 		const y = random(yBuffer, height - yBuffer);
111 		const newPoint = createVector(x, y);
112 		myPoints.push(newPoint);
113 		// points.push(newPoint);
114 	}
115 
116 }
117 
118 /**
119 *   p5.js draw function, is run every frame to create the desired animation
120 */
121 function draw() {
122 	background(100);
123 	drawPoints(myPoints);
124 	let hull = grahamScan(myPoints);
125 	drawHull(hull, 50);
126 }
