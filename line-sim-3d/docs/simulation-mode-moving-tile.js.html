"use strict";

/**
 * Simulation namespace object
 */
var Simulation = Simulation || {};

Simulation.Mode = Simulation.Mode || {};

Simulation.Mode.movingTile = class extends Simulation.Mode.Type {
    constructor() {
        super();
        this.tile = World.Tiles.cross.copy();
        this.sensorRadius = 8;
        this.sensor = new Robot.DigitalLightSensor(this.sensorRadius,
            createVector(0,0));

        this.tileX = 0;
        this.tileXInc = 1;
    }

    update() {
        if(this.tileX < 0 || this.tileX + World.gridSize > width) {
            this.tileXInc *= -1;
        }
        this.tileX += this.tileXInc;

        const tileY = height / 2 - World.gridSize / 2;
        this.tile.setPos(createVector(this.tileX, tileY));
        this.tile.draw();

        this.sensor.setPos(createVector(mouseX, mouseY));
        this.sensor.setRadius(this.sensorRadius);

        const brightness = this.sensor.digitalRead(this.tile);

        if(brightness < 1) {
            console.log(brightness);
        }
        const colorVal = floor(brightness * 255);

        push();
        fill(colorVal);
        strokeWeight(1);
        stroke(127, 0, 30);
        const ellipseSize = 2 * this.sensorRadius;
        ellipse(mouseX, mouseY, ellipseSize, ellipseSize);
        pop();
    }
};
