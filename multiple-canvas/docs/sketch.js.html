/**
*   Function for the first instance of the p5js object.
*/
let sketch1 = function(p) {

  let x = 325;
  let y = 100;
  let vx = -10;

  p.setup = function() {
  	let canvas = p.createCanvas(600, 200);//, p.WEBGL);
  	canvas.parent('sketch1');
  };

  p.draw = function() {
    p.background(0);
    p.translate(0, 0, -p.width/2);
    p.fill(255);
    p.rect(x, y, 50, 50);

    x += vx;
    if(x + 50 > p.width) {
    	vx *= -1;
    }

    if(x < 0) {
    	vx *= -1;
    }

  };
};
